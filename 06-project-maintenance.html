Common Updates

Adding Data Persistence

Install AsyncStorage:

npx expo install @react-native-async-storage/async-storage

Update your main screen:

import AsyncStorage from '@react-native-async-storage/async-storage';

const STORAGE_KEY = 'do_meows_tasks';

export default function HomeScreen() {
  const [tasks, setTasks] = useState<TaskType[]>([]);
  const [isLoading, setIsLoading] = useState(true);

  // Load tasks on app start
  useEffect(() => {
    const loadTasks = async () => {
      try {
        const saved = await AsyncStorage.getItem(STORAGE_KEY);
        if (saved) {
          setTasks(JSON.parse(saved));
        }
      } catch (error) {
        console.error('Failed to load tasks:', error);
      } finally {
        setIsLoading(false);
      }
    };
    loadTasks();
  }, []);

  // Save tasks whenever they change
  useEffect(() => {
    if (!isLoading) {
      AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(tasks));
    }
  }, [tasks, isLoading]);

  // ... rest of component
}

Using FlatList Instead of ScrollView

For better performance with many tasks:

// Before (current code)
<ScrollView>
  {tasks.map((task) => (
    <TaskItem key={task.id} task={task} />
  ))}
</ScrollView>

// After (better for long lists)
<FlatList
  data={tasks}
  keyExtractor={(item) => item.id}
  renderItem={({ item }) => (
    <TaskItem task={item} />
  )}
  ListEmptyComponent={<Text>No tasks yet!</Text>}
/>

Debugging Tips

Common Errors

"Text strings must be rendered within a <Text> component"

‚ùå <View>Hello World</View>

‚úÖ <View><Text>Hello World</Text></View>

"Each child in a list should have a unique 'key' prop"

‚ùå {items.map(item => <Text>{item.name}</Text>)}

‚úÖ {items.map(item => <Text key={item.id}>{item.name}</Text>)}

"Cannot read property 'X' of undefined"

‚ùå <Text>{user.name}</Text> (might crash if user is null)

‚úÖ <Text>{user?.name}</Text> (safe with optional chaining)
